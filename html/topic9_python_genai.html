Context Managers (Resource Cleanup)

from contextlib import contextmanager
import sqlite3

@contextmanager
def get_db_connection(db_path: str):
    """Safe database connection — always closes even on exception."""
    conn = sqlite3.connect(db_path)
    try:
        yield conn
        conn.commit()
    except Exception:
        conn.rollback()
        raise
    finally:
        conn.close()

with get_db_connection("agents.db") as conn:
    cursor = conn.execute("SELECT * FROM sessions")
    results = cursor.fetchall()

Decorators (Retry, Caching, Tracing)

import time
import functools
from typing import Callable, TypeVar

T = TypeVar("T")

def retry_on_rate_limit(max_retries: int = 3, base_delay: float = 1.0):
    """Decorator: retry on OpenAI rate limit errors with exponential backoff."""
    def decorator(func: Callable) -> Callable:
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            last_error = None
            for attempt in range(max_retries):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if "rate_limit" in str(e).lower() or "429" in str(e):
                        delay = base_delay * (2 ** attempt)
                        print(f"Rate limited. Retrying in {delay}s... (attempt {attempt+1}/{max_retries})")
                        time.sleep(delay)
                        last_error = e
                    else:
                        raise  # Re-raise non-rate-limit errors immediately
            raise last_error
        return wrapper
    return decorator

@retry_on_rate_limit(max_retries=3)
def call_llm(prompt: str) -> str:
    return client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}]
    ).choices[0].message.content

3. Pydantic for Data Validation

Why Pydantic in GenAI?

LLMs output text. You need structured, validated Python objects. Pydantic bridges the gap — it parses, validates, and type-checks LLM outputs, API request bodies, and agent configurations.

from pydantic import BaseModel, Field, field_validator, model_validator
from typing import Optional
from datetime import datetime

class AgentConfig(BaseModel):
    model: str = Field(default="gpt-4o-mini", pattern=r"^gpt-|^claude-|^gemini-")
    temperature: float = Field(default=0.0, ge=0.0, le=2.0)
    max_tokens: int = Field(default=1000, gt=0, le=16000)
    max_iterations: int = Field(default=5, ge=1, le=20)
    system_prompt: str = Field(..., min_length=10, max_length=4000)
    tools_enabled: list[str] = Field(default_factory=list)

    @field_validator("tools_enabled", mode="before")  # Pydantic v2: field_validator replaces validator
    @classmethod
    def validate_tool_names(cls, tools):
        allowed = {"search_web", "python_repl", "knowledge_base", "calculator"}
        for tool in tools:
            if tool not in allowed:
                raise ValueError(f"Unknown tool: {tool}. Allowed: {allowed}")
        return tools

# LLM structured output parsing
class SearchResult(BaseModel):
    title: str
    url: str
    summary: str = Field(max_length=500)
    relevance_score: float = Field(ge=0.0, le=1.0)

class SearchResponse(BaseModel):
    query: str
    results: list[SearchResult]
    total_found: int

    @model_validator(mode="after")
    def validate_counts(self):
        if len(self.results) > self.total_found:
            raise ValueError("results count cannot exceed total_found")
        return self

# Use with OpenAI structured outputs (gpt-4o supports native JSON mode)
response = client.beta.chat.completions.parse(
    model="gpt-4o",
    messages=[{"role": "user", "content": "Search results for 'Python async programming'"}],
    response_format=SearchResponse
)
search = response.choices[0].message.parsed
print(f"Found {search.total_found} results")
for r in search.results:
    print(f"  [{r.relevance_score:.2f}] {r.title}")

4. FastAPI for LLM APIs

from fastapi import FastAPI, HTTPException, Depends, Header
from pydantic import BaseModel
from typing import AsyncIterator
from openai import AsyncOpenAI
import asyncio

app = FastAPI(title="GenAI API", version="1.0.0")
async_client = AsyncOpenAI()

# Request/Response schemas
class ChatRequest(BaseModel):
    message: str = Field(..., min_length=1, max_length=2000)
    session_id: str = Field(default="default", pattern=r"^[a-zA-Z0-9_-]{1,64}$")
    model: str = Field(default="gpt-4o-mini")

class ChatResponse(BaseModel):
    reply: str
    tokens_used: int
    model: str

# API key authentication
async def verify_api_key(x_api_key: str = Header(...)):
    """Simple API key auth — use OAuth2/JWT in production."""
    if x_api_key != "expected-key":
        raise HTTPException(status_code=401, detail="Invalid API key")
    return x_api_key

@app.post("/chat", response_model=ChatResponse)
async def chat(request: ChatRequest, _=Depends(verify_api_key)):
    """Non-streaming chat endpoint."""
    response = await async_client.chat.completions.create(
        model=request.model,
        messages=[{"role": "user", "content": request.message}],
        max_tokens=1000
    )
    return ChatResponse(
        reply=response.choices[0].message.content,
        tokens_used=response.usage.total_tokens,
        model=request.model
    )

from fastapi.responses import StreamingResponse

@app.post("/chat/stream")
async def chat_stream(request: ChatRequest, _=Depends(verify_api_key)):
    """Streaming chat endpoint — sends chunks as SSE."""

    async def generate() -> AsyncIterator[str]:
        stream = await async_client.chat.completions.create(
            model=request.model,
            messages=[{"role": "user", "content": request.message}],
            stream=True
        )
        async for chunk in stream:
            delta = chunk.choices[0].delta.content
            if delta:
                yield f"data: {delta}\n\n"
        yield "data: [DONE]\n\n"

    return StreamingResponse(generate(), media_type="text/event-stream")

6. Async Python for High-Performance Pipelines

LLM API calls are I/O-bound and can take 1–10 seconds each. Async Python lets you run multiple calls concurrently without threads.

import asyncio
from openai import AsyncOpenAI

async_client = AsyncOpenAI()

async def summarize(text: str) -> str:
    response = await async_client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": f"Summarize in one sentence: {text}"}]
    )
    return response.choices[0].message.content

# Run 5 summaries concurrently (not sequentially!)
async def summarize_all(texts: list[str]) -> list[str]:
    tasks = [summarize(text) for text in texts]
    return await asyncio.gather(*tasks, return_exceptions=True)

texts = ["Article 1 content...", "Article 2 content...", "Article 3 content...",
         "Article 4 content...", "Article 5 content..."]

# Sequential: ~15 seconds (5 × 3s each)
# Concurrent: ~3 seconds (all at once)
summaries = asyncio.run(summarize_all(texts))

# With rate limiting (semaphore prevents overwhelming the API)
async def summarize_with_limit(texts: list[str], max_concurrent: int = 5) -> list[str]:
    semaphore = asyncio.Semaphore(max_concurrent)

    async def bounded_summarize(text: str) -> str:
        async with semaphore:
            return await summarize(text)

    return await asyncio.gather(*[bounded_summarize(t) for t in texts])

7. Complete Pipeline: LLM + RAG + Agent

"""
Complete GenAI pipeline combining all concepts:
- Pydantic validation
- OpenAI SDK (embeddings + chat)
- FAISS vector store
- FastAPI serving
- Async execution
"""
import asyncio
import os
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel, Field
from openai import AsyncOpenAI
from langchain_openai import OpenAIEmbeddings
from langchain_community.vectorstores import FAISS
from langchain_community.document_loaders import TextLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from dotenv import load_dotenv

load_dotenv()

app = FastAPI(title="RAG + Agent Pipeline")
async_client = AsyncOpenAI()

# Build RAG on startup
# Note: @app.on_event is deprecated since FastAPI 0.93; prefer lifespan context manager in new projects
@app.on_event("startup")
async def build_rag():
    loader = TextLoader("knowledge_base.txt")
    splitter = RecursiveCharacterTextSplitter(chunk_size=500, chunk_overlap=50)
    docs = splitter.split_documents(loader.load())
    embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
    app.state.vectorstore = FAISS.from_documents(docs, embeddings)

class QueryRequest(BaseModel):
    question: str = Field(..., min_length=3, max_length=500)
    top_k: int = Field(default=3, ge=1, le=10)

class QueryResponse(BaseModel):
    answer: str
    sources: list[str]
    tokens_used: int

@app.post("/query", response_model=QueryResponse)
async def query(request: QueryRequest):
    # 1. Retrieve relevant context
    docs = app.state.vectorstore.similarity_search(request.question, k=request.top_k)
    context = "\n\n".join(d.page_content for d in docs)
    sources = list({d.metadata.get("source", "unknown") for d in docs})

    # 2. Generate grounded answer
    response = await async_client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {"role": "system", "content":
             "Answer based ONLY on the provided context. If not in context, say 'I don't know.'"},
            {"role": "user", "content": f"Context:\n{context}\n\nQuestion: {request.question}"}
        ],
        temperature=0,
        max_tokens=400
    )
    return QueryResponse(
        answer=response.choices[0].message.content,
        sources=sources,
        tokens_used=response.usage.total_tokens
    )
