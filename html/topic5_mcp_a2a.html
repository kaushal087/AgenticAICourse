1. Model Context Protocol (MCP)

The Problem MCP Solves

Before MCP, every AI agent needed custom integration code for each tool or data source — GitHub, Slack, databases, filesystems. An agent using 10 tools needed 10 different integrations. This created an explosion of fragile, unmaintainable connector code.

MCP (introduced by Anthropic, November 2024) standardizes how AI agents communicate with tools and data sources using a single, unified protocol.

MCP Architecture

┌─────────────────────────────────────────────────────────┐
│                    MCP Client (Agent)                   │
│   "I need to read a file, query a DB, search the web"  │
└──────────────────────┬──────────────────────────────────┘
                       │  MCP Protocol (JSON-RPC 2.0)
                       │  over stdio / HTTP / WebSocket
       ┌───────────────┼───────────────┐
       ▼               ▼               ▼
┌──────────────┐ ┌────────────┐ ┌──────────────┐
│  MCP Server  │ │ MCP Server │ │  MCP Server  │
│  (Filesystem)│ │ (GitHub)   │ │  (PostgreSQL)│
└──────────────┘ └────────────┘ └──────────────┘

Building an MCP Server in Python

# Install: pip install mcp
from mcp.server import Server
from mcp.server.stdio import stdio_server
from mcp import types
import asyncio

server = Server("my-knowledge-server")

# Define a resource (data the agent can read)
@server.list_resources()
async def list_resources() -> list[types.Resource]:
    return [
        types.Resource(
            uri="knowledge://company-docs",
            name="Company Documentation",
            description="Internal company policies and procedures",
            mimeType="text/plain"
        )
    ]

@server.read_resource()
async def read_resource(uri: str) -> str:
    if uri == "knowledge://company-docs":
        with open("knowledge_base.txt") as f:
            return f.read()
    raise ValueError(f"Unknown resource: {uri}")

# Define a tool (action the agent can call)
@server.list_tools()
async def list_tools() -> list[types.Tool]:
    return [
        types.Tool(
            name="search_documents",
            description="Search internal documents for relevant information",
            inputSchema={
                "type": "object",
                "properties": {
                    "query": {"type": "string", "description": "Search query"}
                },
                "required": ["query"]
            }
        )
    ]

@server.call_tool()
async def call_tool(name: str, arguments: dict) -> list[types.TextContent]:
    if name == "search_documents":
        query = arguments["query"]
        # Perform vector search
        results = rag_search(query)
        return [types.TextContent(type="text", text=results)]

async def main():
    async with stdio_server() as (read_stream, write_stream):
        await server.run(read_stream, write_stream, server.create_initialization_options())

asyncio.run(main())

Connecting an Agent to MCP Servers

from langchain_mcp_adapters.client import MultiServerMCPClient
from langchain.agents import create_openai_functions_agent, AgentExecutor
from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
import asyncio

async def run_mcp_agent():
    # Connect to multiple MCP servers simultaneously
    async with MultiServerMCPClient({
        "knowledge": {
            "command": "python",
            "args": ["knowledge_server.py"],
            "transport": "stdio"
        },
        "github": {
            "url": "http://localhost:8001/mcp",
            "transport": "http"
        }
    }) as client:
        # All MCP tools auto-discovered — no manual integration!
        tools = await client.get_tools()
        print(f"Discovered {len(tools)} tools: {[t.name for t in tools]}")

        llm = ChatOpenAI(model="gpt-4o", temperature=0)
        prompt = ChatPromptTemplate.from_messages([
            ("system", "You have access to company knowledge and GitHub. Use them as needed."),
            ("human", "{input}"),
            MessagesPlaceholder("agent_scratchpad"),
        ])
        agent = create_openai_functions_agent(llm, tools, prompt)
        executor = AgentExecutor(agent=agent, tools=tools, verbose=True)
        result = await executor.ainvoke({"input": "Find our API documentation and create a GitHub issue for missing endpoint docs"})
        print(result["output"])

asyncio.run(run_mcp_agent())

2. Agent-to-Agent (A2A) Protocol

What is A2A?

The A2A Protocol (Google, April 2025) defines how AI agents discover, communicate with, and delegate tasks to each other across different systems, frameworks, and vendors.

AspectMCPA2A
PurposeAgent ↔ Tool/Resource communicationAgent ↔ Agent communication
Typical relationshipAgent calls a toolAgent delegates to another agent
ProtocolJSON-RPC 2.0HTTP/HTTPS + JSON
Capability discoveryServer exposes tools listAgent Card (/.well-known/agent.json)
CreatorAnthropic (Nov 2024)Google (Apr 2025)

A2A Agent Card

Each A2A agent publishes a self-description at a well-known URL — like a business card for agents:

3. Building A2A Communication with LangGraph

Agent-to-Agent Communication via Shared State

from typing import TypedDict, Annotated
from langgraph.graph import StateGraph, END
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, AIMessage
import operator

class MultiAgentState(TypedDict):
    task: str
    plan: str
    research_results: str
    analysis: str
    final_report: str
    current_agent: str
    # Message log for full A2A trace
    message_log: Annotated[list[dict], operator.add]

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

def planner_agent(state: MultiAgentState) -> MultiAgentState:
    """Orchestrator: Break task into subtasks."""
    response = llm.invoke([
        HumanMessage(content=f"""You are a Planner agent.
Task: {state['task']}
Create a step-by-step research plan with 3-5 specific subtasks.""")
    ])
    return {
        "plan": response.content,
        "current_agent": "researcher",
        "message_log": [{"from": "planner", "to": "researcher",
                          "content": f"Plan created: {response.content[:100]}..."}]
    }

def researcher_agent(state: MultiAgentState) -> MultiAgentState:
    """Specialist: Execute research based on the plan."""
    response = llm.invoke([
        HumanMessage(content=f"""You are a Research agent.
Plan: {state['plan']}
Execute the research plan. Provide detailed findings for each subtask.""")
    ])
    return {
        "research_results": response.content,
        "current_agent": "analyst",
        "message_log": [{"from": "researcher", "to": "analyst",
                          "content": "Research complete, forwarding results."}]
    }

def analyst_agent(state: MultiAgentState) -> MultiAgentState:
    """Specialist: Analyze and synthesize research."""
    response = llm.invoke([
        HumanMessage(content=f"""You are an Analyst agent.
Research: {state['research_results']}
Evaluate the quality and completeness. Identify key insights and gaps.""")
    ])
    return {
        "analysis": response.content,
        "current_agent": "writer",
        "message_log": [{"from": "analyst", "to": "writer",
                          "content": "Analysis complete. Writing final report."}]
    }

def writer_agent(state: MultiAgentState) -> MultiAgentState:
    """Specialist: Write final report."""
    response = llm.invoke([
        HumanMessage(content=f"""You are a Writer agent.
Research: {state['research_results']}
Analysis: {state['analysis']}
Write a professional 400-word report with executive summary and recommendations.""")
    ])
    return {
        "final_report": response.content,
        "current_agent": "done",
        "message_log": [{"from": "writer", "to": "output",
                          "content": "Report complete."}]
    }

# Build the graph
workflow = StateGraph(MultiAgentState)
workflow.add_node("planner", planner_agent)
workflow.add_node("researcher", researcher_agent)
workflow.add_node("analyst", analyst_agent)
workflow.add_node("writer", writer_agent)

workflow.set_entry_point("planner")
workflow.add_edge("planner", "researcher")
workflow.add_edge("researcher", "analyst")
workflow.add_edge("analyst", "writer")
workflow.add_edge("writer", END)

graph = workflow.compile()

# Run
result = graph.invoke({
    "task": "Research the impact of multi-agent systems on enterprise software development",
    "plan": "", "research_results": "", "analysis": "",
    "final_report": "", "current_agent": "planner", "message_log": []
})

print("=== Final Report ===")
print(result["final_report"])
print("\n=== Agent Communication Log ===")
for msg in result["message_log"]:
    print(f"  {msg['from']} → {msg['to']}: {msg['content'][:80]}")
