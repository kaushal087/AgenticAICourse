    subgraph Ingestion["📥 Ingestion (one-time)"]

        S["RecursiveCharacter\nTextSplitter\n(chunks ~500 chars)"]

    subgraph Retrieval["🔍 Retrieval (per query)"]
        Q[/"🔎 Query String"/]
        QE["OpenAI Embeddings\n(same model)"]
        SS["Similarity Search\ncosine distance"]
        TK["Top-K Chunks\n(k=3)"]
        CTX["📋 Context\nfor LLM"]

        Q --> QE --> SS
        I --> SS --> TK --> CTX
    end

        CTX --> LLM --> ANS
    end
