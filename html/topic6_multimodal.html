1. Multimodal AI: Beyond Text

The Modality Spectrum

ModalityInput ExamplesOutput ExamplesKey Model
TextUser queries, documentsAnswers, code, summariesGPT-4o, Claude 3.5
AudioVoice commands, call recordingsSpoken responses, transcriptsWhisper, GPT-4o Audio
VisionScreenshots, diagrams, photosDescriptions, data extractionGPT-4o Vision, Claude 3
VideoRecordings, screen capturesSummaries, event detectionGemini 1.5 Pro
DocumentsPDFs, spreadsheetsExtraction, Q&AGPT-4o + vision

Pipeline Architectures

Classic Voice Pipeline (Cascaded):
Audio Input → ASR (Whisper) → Text → LLM → Text → TTS → Audio Output
    └── Each step is independent; errors compound

Native Multimodal Pipeline:
Audio Input ─────────────────────────────────────────▶ GPT-4o Audio
                                                           │
                               ┌───────────────────────────┤
                               ▼                           ▼
                          Audio Output               Text Output

Key trade-off: Cascaded pipelines give you more control and debuggability. Native multimodal (GPT-4o Realtime) has lower latency and preserves emotional/prosodic cues in speech.

2. Automatic Speech Recognition: OpenAI Whisper

Whisper Overview

Whisper is OpenAI's open-source ASR model, trained on 680,000 hours of multilingual audio. It handles background noise, accents, and technical vocabulary remarkably well.

Transcribing Audio

# Option 1: Local Whisper (pip install openai-whisper)
import whisper

model = whisper.load_model("base")  # or "small", "medium", "large-v3"

# Transcribe a file
result = model.transcribe("audio.mp3")
print(result["text"])
print(f"Language detected: {result['language']}")

# With verbosity: timestamps per segment
result = model.transcribe("meeting.mp3", verbose=False)
for segment in result["segments"]:
    start = segment["start"]
    end = segment["end"]
    text = segment["text"]
    print(f"[{start:.1f}s → {end:.1f}s] {text}")

# Option 2: OpenAI API (faster, no GPU needed)
from openai import OpenAI
client = OpenAI()

with open("audio.mp3", "rb") as audio_file:
    transcription = client.audio.transcriptions.create(
        model="whisper-1",
        file=audio_file,
        language="en",     # Optional: specify language for better accuracy
        timestamp_granularities=["segment"]  # Get timestamps
    )
print(transcription.text)

Real-Time Streaming Transcription

import pyaudio
import numpy as np
import whisper
import queue
import threading

model = whisper.load_model("base")
audio_queue = queue.Queue()

def record_audio(chunk_duration=3):
    """Record audio in 3-second chunks."""
    RATE = 16000
    CHUNK = 1024
    p = pyaudio.PyAudio()
    stream = p.open(format=pyaudio.paFloat32, channels=1,
                    rate=RATE, input=True, frames_per_buffer=CHUNK)
    print("Listening... (Ctrl+C to stop)")
    while True:
        frames = []
        for _ in range(0, int(RATE / CHUNK * chunk_duration)):
            data = stream.read(CHUNK)
            frames.append(np.frombuffer(data, dtype=np.float32))
        audio_chunk = np.concatenate(frames)
        audio_queue.put(audio_chunk)

def transcribe_chunks():
    """Process audio chunks as they arrive."""
    while True:
        audio_chunk = audio_queue.get()
        result = model.transcribe(audio_chunk, fp16=False)
        text = result["text"].strip()
        if text:
            print(f"Heard: {text}")
            # Feed to agent
            process_with_agent(text)

# Run both threads concurrently
threading.Thread(target=record_audio, daemon=True).start()
transcribe_chunks()

4. GPT-4o Vision: Seeing the World

from openai import OpenAI
import base64

client = OpenAI()

def encode_image(image_path: str) -> str:
    with open(image_path, "rb") as f:
        return base64.b64encode(f.read()).decode("utf-8")

def analyze_image(image_path: str, question: str) -> str:
    """Ask GPT-4o a question about an image."""
    image_data = encode_image(image_path)

    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[
            {
                "role": "user",
                "content": [
                    {
                        "type": "image_url",
                        "image_url": {
                            "url": f"data:image/jpeg;base64,{image_data}",
                            "detail": "high"  # "low" for speed, "high" for detail
                        }
                    },
                    {"type": "text", "text": question}
                ]
            }
        ],
        max_tokens=500
    )
    return response.choices[0].message.content

# Use cases
print(analyze_image("screenshot.png", "What errors do you see in this code screenshot?"))
print(analyze_image("chart.png", "Describe the trend shown in this chart and identify the peak."))

5. OpenAI Realtime API: Native Audio-to-Audio

# pip install openai websockets pyaudio
import asyncio
import json
import websockets
import pyaudio
import base64

REALTIME_API_URL = "wss://api.openai.com/v1/realtime?model=gpt-4o-realtime-preview-2024-10-01"

async def run_realtime_agent():
    headers = {
        "Authorization": f"Bearer {os.environ['OPENAI_API_KEY']}",
        "OpenAI-Beta": "realtime=v1"
    }

    async with websockets.connect(REALTIME_API_URL, extra_headers=headers) as ws:
        # Configure session
        await ws.send(json.dumps({
            "type": "session.update",
            "session": {
                "modalities": ["text", "audio"],
                "instructions": "You are a helpful voice assistant. Be concise. Use natural speech.",
                "voice": "alloy",
                "input_audio_format": "pcm16",
                "output_audio_format": "pcm16",
                "input_audio_transcription": {"model": "whisper-1"},
                "turn_detection": {
                    "type": "server_vad",      # Server-side voice activity detection
                    "threshold": 0.5,
                    "silence_duration_ms": 800
                },
                "tools": [
                    {
                        "type": "function",
                        "name": "get_weather",
                        "description": "Get current weather for a city",
                        "parameters": {
                            "type": "object",
                            "properties": {
                                "city": {"type": "string"}
                            }
                        }
                    }
                ]
            }
        }))

        # Stream audio input and handle responses
        async for message in ws:
            event = json.loads(message)
            event_type = event.get("type")

            if event_type == "conversation.item.input_audio_transcription.completed":
                print(f"User said: {event['transcript']}")

            elif event_type == "response.audio.delta":
                # Play audio chunk immediately (streaming)
                audio_bytes = base64.b64decode(event["delta"])
                play_audio_chunk(audio_bytes)

            elif event_type == "response.function_call_arguments.done":
                # Handle tool calls
                func_name = event["name"]
                args = json.loads(event["arguments"])
                result = handle_tool_call(func_name, args)
                # Send result back
                await ws.send(json.dumps({
                    "type": "conversation.item.create",
                    "item": {
                        "type": "function_call_output",
                        "call_id": event["call_id"],
                        "output": result
                    }
                }))
                await ws.send(json.dumps({"type": "response.create"}))

asyncio.run(run_realtime_agent())

6. Complete Multimodal Agent Pipeline

import asyncio
from openai import OpenAI
import whisper
import pyaudio
import pygame
import numpy as np

client = OpenAI()
whisper_model = whisper.load_model("base")

class MultimodalAgent:
    """Voice + Vision + Text agent."""

    def __init__(self):
        self.conversation_history = []

    def transcribe_voice(self, audio_array: np.ndarray) -> str:
        result = whisper_model.transcribe(audio_array, fp16=False)
        return result["text"].strip()

    def analyze_image(self, image_path: str) -> str:
        """Returns a description of the image."""
        with open(image_path, "rb") as f:
            image_data = base64.b64encode(f.read()).decode()
        response = client.chat.completions.create(
            model="gpt-4o",
            messages=[{"role": "user", "content": [
                {"type": "image_url", "image_url": {"url": f"data:image/jpeg;base64,{image_data}"}},
                {"type": "text", "text": "Describe this image concisely for an AI agent."}
            ]}]
        )
        return response.choices[0].message.content

    def reason(self, text_input: str, image_context: str = None) -> str:
        messages = [{"role": "system", "content":
            "You are a helpful multimodal assistant. Be concise and actionable."}]
        messages.extend(self.conversation_history)
        if image_context:
            text_input = f"[Image context: {image_context}]\n\nUser: {text_input}"
        messages.append({"role": "user", "content": text_input})

        response = client.chat.completions.create(model="gpt-4o-mini", messages=messages)
        answer = response.choices[0].message.content
        self.conversation_history.append({"role": "user", "content": text_input})
        self.conversation_history.append({"role": "assistant", "content": answer})
        return answer

    def speak(self, text: str):
        response = client.audio.speech.create(model="tts-1", voice="nova", input=text)
        response.stream_to_file("/tmp/response.mp3")
        pygame.mixer.init()
        pygame.mixer.music.load("/tmp/response.mp3")
        pygame.mixer.music.play()
        while pygame.mixer.music.get_busy():
            pygame.time.wait(50)

agent = MultimodalAgent()

# Example: user speaks + shares a screenshot
user_voice = record_3_seconds()                         # returns np.ndarray
user_text = agent.transcribe_voice(user_voice)          # "What's wrong with this code?"
image_desc = agent.analyze_image("screenshot.png")      # describes the code screenshot
answer = agent.reason(user_text, image_context=image_desc)
agent.speak(answer)                                     # agent responds out loud
